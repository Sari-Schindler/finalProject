class SpyHoldStrategy extends Strategy {
   constructor() {
       super('SPY', 100); // stockSymbol is 'SPY', quantity is $100
   }
   firstDayActions() {
       console.log("Buying ", this.quantity, "worth of ", this.stockSymbol, " and holding for 10 months.");
   }
   dayActions() {
       if (this.portfolio.isFirstDayOfMonth()) {
           this.portfolio.buyStocksByDollars(this.quantity);
       }
       if (this.portfolio.convertStringToDate(this.portfolio.currentDate).getMonth() === this.portfolio.convertStringToDate(this.portfolio.currentDate).getMonth() + 10) {
           this.portfolio.sellAll();
           console.log("Sold all shares of ", this.stockSymbol, " after holding for 10 months.");
       }
   }
}
function main() {
   const myStrategy = new SpyHoldStrategy();
   const myRangesExecutor = new RangesExecutor(stocksData, myStrategy, 100, 300);
   const executionResults = myRangesExecutor.allRangesExecution();
   console.log("execution results:", executionResults);
   fs.writeFile(path.resolve(__dirname, '../data/executionResults.json'),
                JSON.stringify(executionResults),
                 function(err) {
                       if (err) throw err;
                       console.log('complete');
                 });
}
main();
